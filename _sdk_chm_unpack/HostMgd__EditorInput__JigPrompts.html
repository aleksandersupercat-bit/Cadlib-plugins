Описание
Примечания

Класс предназначен для взаимодействия с пользователем во время использования технологии Jig и предоставляет возможность запрашивать у пользователя данные разного типа для построения или изменения графических объектов. 

 

Пример

 

В примере приведена команда JigRotateScaleMove, с помощью которой можно вращать, масштабировать(уменьшать/увеличивать) и перемещать выбранный примитив или группу примитивов с помощью технологии Jig. Для вращения, масштабирования и перемещения используются методы JigPrompts.AcquireAngle(JigPromptAngleOptions options), JigPrompts.AcquireDistance(JigPromptDistanceOptions options), JigPrompts.AcquirePoint(JigPromptPointOptions options) соответственно. 

Логика работы команды JigRotateScaleMove реализована в классе RotateScaleMove. В этом классе применяются рассматриваемые в настоящем разделе методы.

using HostMgd.ApplicationServices; using HostMgd.EditorInput; using Teigha.Runtime; using Teigha.DatabaseServices; using System.Collections.Generic; using Teigha.Geometry; using Teigha.GraphicsInterface; using System; class RotateScaleMove : DrawJig { //Переменная entities для коллекции примитивов, которые будут изменяться public List<Entity> entities = new List<Entity>(); //Переменная KeyWord для хранения выбранного над примитивами действия public string keyWord; //Переменная basePoint для хранения выбранной точки отсчета для изменений public Point3d basePoint; //Переменная angle для хранения угла поворота примитивов public double angle; //Переменная scale для хранения коэффициента масштабирования примитивов public double scale; //Переменная newPoint для хранения точки, в которую будут перемещены примитивы public Point3d newPoint; //Переменная currentUCS для хранения системы координат, в которой созданы примитивы Matrix3d currentUCS; //Переменная transformationMatrix для хранения матрицы преобразования примитивов public Matrix3d transformationMatrix; //Для изменения выбранных пользователем примитивов в экземпляр класса необходимо передать: // - систему координат примитивов matrix // - выбранную точку отсчета для изменения _basePoint // - выбранное действие над примитивами kW public RotateScaleMove(Matrix3d matrix, Point3d _basePoint,string kW) { this.currentUCS = matrix; this.basePoint = _basePoint; transformationMatrix = new Matrix3d(); keyWord = kW; } protected override SamplerStatus Sampler(JigPrompts prompts) { switch (keyWord) { case "ВРАЩЕНИЕ": { //Определение экземпляра класса JigPromptAngleOptions и его свойств: // - приглашающее сообщение пользователю // - определение точки, относительно которой будут поворачиваться выбранные объекты JigPromptAngleOptions angleOptions = new JigPromptAngleOptions("Повернуть:"); angleOptions.UseBasePoint = true; angleOptions.BasePoint = basePoint; //Запрос угла поворота у пользователя с применением экземпляра класса // JigPromptAngleOptions PromptDoubleResult angleResult = prompts.AcquireAngle(angleOptions); if (angleResult.Status == PromptStatus.Cancel || angleResult.Status == PromptStatus.Error) return SamplerStatus.Cancel; if (angleResult.Value == angle) return SamplerStatus.NoChange; //Если метод prompt.AcquireAngle(angleOptions) завершил свою работу со статусом PromptStatus.OK, // переменной angle присваивается выбранное пользователем значение angle = angleResult.Value; // Переменной transformationMatrix присваивается матрица вращения transformationMatrix = Matrix3d.Rotation(angleResult.Value, currentUCS.CoordinateSystem3d.Zaxis, basePoint); //Завершение работы метода с дальнейшим вызовом метода WorldDraw() return SamplerStatus.OK; } case "МАСШТАБ": { //Определение экземпляра класса JigPromptDistanceOptions и его свойств JigPromptDistanceOptions distanceOptions = new JigPromptDistanceOptions("Масштабировать:"); distanceOptions.UseBasePoint = true; distanceOptions.BasePoint = basePoint; //Запрос расстояния, которое будет использовано как коэффициент масштабирования с // применением экземпляра класса JigPromptDistanceOptions PromptDoubleResult distanceResult = prompts.AcquireDistance(distanceOptions); if (distanceResult.Status == PromptStatus.Cancel || distanceResult.Status == PromptStatus.Error) return SamplerStatus.Cancel; if (distanceResult.Value == scale) return SamplerStatus.NoChange; //Если метод prompt.AcquireDistance(distanceOptions) завершил свою работу со статусом PromptStatus.OK, // переменной scale присваивается выбранное пользователем значение scale = distanceResult.Value; // Переменной transformationMatrix присваивается матрица масштабирования transformationMatrix = Matrix3d.Scaling(scale, basePoint); return SamplerStatus.OK; } case "ПЕРЕМЕЩЕНИЕ": { //Определение экземпляра класса JigPromptPointOptions и его свойств: JigPromptPointOptions moveOptions = new JigPromptPointOptions("Переместить:"); moveOptions.UseBasePoint = true; moveOptions.BasePoint = basePoint; //Запрос точки, в которую будет перемещена выбранная точка отсчета для изменения примитивов с // применением экземпляра класса JigPromptPointOptions PromptPointResult pointResult = prompts.AcquirePoint(moveOptions); if (pointResult.Status == PromptStatus.Cancel || pointResult.Status == PromptStatus.Error) return SamplerStatus.Cancel; if (pointResult.Value == basePoint) return SamplerStatus.NoChange; //Если метод prompt.AcquirePoint(pointOptions) завершил свою работу со статусом PromptStatus.OK, // переменной newPoint присваивается выбранное пользователем значение newPoint = pointResult.Value; // Переменной transformationMatrix присваивается матрица перемещения transformationMatrix = Matrix3d.Displacement(basePoint.GetVectorTo(pointResult.Value)); return SamplerStatus.OK; } default:break; } return SamplerStatus.OK; } protected override bool WorldDraw(WorldDraw draw) { //Добавление нового преобразования в стек преобразований draw.Geometry.PushModelTransform(transformationMatrix); //Отображение преобразованых объектов foreach (var ent in entities) { draw.Geometry.Draw(ent); } //Возврат объектов в начальное состояние draw.Geometry.PopModelTransform(); return true; } } public partial class Commands { [CommandMethod("JigRotateScaleMove")] public void JigRotateScaleMove() { Document doc = Application.DocumentManager.MdiActiveDocument; Editor ed = doc.Editor; Database db = doc.Database; //Запрос действия для преобразования объекта(ов) PromptStringOptions stringOptions = new PromptStringOptions("Выберите действие [ВРАЩЕНИЕ/МАСШТАБ/ПЕРЕМЕЩЕНИЕ]:"); PromptResult stringResult = ed.GetString(stringOptions); if (stringResult.Status != PromptStatus.OK) return; //Запрос объектов, которые будут преобразованы PromptSelectionOptions selectionOptions = new PromptSelectionOptions(); selectionOptions.MessageForAdding = "Выберите объекты, которые необходимо изменить:"; selectionOptions.MessageForRemoval = "Выберите объекты, которые необходимо убрать из выделения:"; PromptSelectionResult selectionResult = ed.GetSelection(selectionOptions); if (selectionResult.Status != PromptStatus.OK) return; //Запрос точки, относительно которой будет выполнено преобразование PromptPointOptions pointOptions = new PromptPointOptions("Выберите базовую точку:"); PromptPointResult pointResult = ed.GetPoint(pointOptions); if (pointResult.Status != PromptStatus.OK) return; //Определение экземпляра класса-наследника DrawJig, который преобразует выбранные объекты по технологии Jig RotateScaleMove rotateScaleMove = new RotateScaleMove(ed.CurrentUserCoordinateSystem, pointResult.Value,stringResult.StringResult); //Начало транзакции с базой данных документа using (Transaction trans = db.TransactionManager.StartTransaction()) { //Получение Id примитивов из базы данных foreach (ObjectId id in selectionResult.Value.GetObjectIds()) { //Открытие каждого выбранного примитива для изменений Entity ent = (Entity)trans.GetObject(id, OpenMode.ForWrite); //Добавление каждого выбранного пользователем примитива в коллекцию класса RotateScaleMove entities rotateScaleMove.entities.Add(ent); } //Вызов метода Editor.Drag() для изменения выбранных примитивов по технологии Jig PromptResult prompt = ed.Drag(rotateScaleMove); //Если метод успешно выполнен, последнее преобразование сохраняется в базе данных документа if (prompt.Status == PromptStatus.OK) { foreach (Entity ent in rotateScaleMove.entities) ent.TransformBy(rotateScaleMove.transformationMatrix); trans.Commit(); } //Если метод выполнен с ошибкой или была отмена действий, изменения не сохраняются в базе данных else trans.Abort(); } //Вывод сообщения о выполненном преобразовании. switch (rotateScaleMove.keyWord) { case "ВРАЩЕНИЕ": { //При повороте объекта(ов) значение угла поворота переводится из радиан в градусы ed.WriteMessage("Поворот на {0} градусов", rotateScaleMove.angle*180/Math.PI); break; } case "МАСШТАБ": { ed.WriteMessage("Масштабирование с коэффициентом {0}", rotateScaleMove.scale); break; } case "ПЕРЕМЕЩЕНИЕ": { ed.WriteMessage("Перемещение из точки ({0},{1},{2}) в точку ({3},{4},{5})", rotateScaleMove.basePoint.X, rotateScaleMove.basePoint.Y, rotateScaleMove.basePoint.Z, rotateScaleMove.newPoint.X, rotateScaleMove.newPoint.Y, rotateScaleMove.newPoint.Z); break; } } } }

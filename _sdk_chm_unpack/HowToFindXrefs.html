Как просмотреть список внешних ссылок dwg-документа
Послать отзыв

В этой статье приведен пример того, как просмотреть все внешние зависимости, которые присутстсвуют в dwg-документе.

Описание

 

Копировать Код
[Teigha.Runtime.CommandMethod("LinksListReading")] public void LinksListReading() { // Получение ссылки на активный документ HostMgd.ApplicationServices.Document doc = HostMgd.ApplicationServices.Application.DocumentManager.MdiActiveDocument; // Получение ссылки на редактор докумена HostMgd.EditorInput.Editor ed = doc.Editor; // Получение ссылки на базу данных документа Teigha.DatabaseServices.Database db = doc.Database; // Записывыаем ссылку на менеджер зависимостей файла в переменную Teigha.DatabaseServices.FileDependencyManager dependencyManager = db.FileDependencyManager; // Перебирем в цикле зависимости документа for(int i=1;i<=dependencyManager.CountEntries;i++) { // Записываем в переменную информацию о зависимости Teigha.DatabaseServices.FileDependencyInfo info=dependencyManager.GetEntry(i); // Выводим в командную строку названиефайла, с которым у данного dwg-документа есть зависимость // Также выводим тип зависимости, который записан в свойстве Feature ed.WriteMessage($"FileName {info.FileName}, Feature {info.Feature}"); } }

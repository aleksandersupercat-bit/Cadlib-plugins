Описание

 

В свойстве хранится цвет текста, относящегося к размерам, в виде типа данных Teigha.Colors.Color. Помимо различных цветов для этого свойства доступны значения «BYBLOCK» и «BYLAYER». Значение по умолчанию – «BYBLOCK». 

Свойство также доступно через системную переменную DIMCLRT и пользовательское меню в окне изменения/создания размерного стиля: 

 

При изменении системной переменной DIMCLRT будет создан новый размерный стиль под названием <переопределение> на основе текущего с новым значением DIMCLRT.

Пример изменения цвета текста, относящегося к размерам (значения свойства Dimclrt) в размерном стиле Standard:

Копировать Код
public partial class Commands { [Teigha.Runtime.CommandMethod("DimclrtChange")] public void DimclrtStandardDimStyle() { // Получение ссылки на активный документ HostMgd.ApplicationServices.Document doc = HostMgd.ApplicationServices.Application.DocumentManager.MdiActiveDocument; // Получение ссылки на редактор документа HostMgd.EditorInput.Editor ed = doc.Editor; // Получение ссылки на базу данных документа Teigha.DatabaseServices.Database db = doc.Database; // Начало транзакции с базой данных документа using (Teigha.DatabaseServices.Transaction trans = db.TransactionManager.StartTransaction()) { // Открываем таблицу размерных стилей для чтения Teigha.DatabaseServices.DimStyleTable dimStyleTable = trans.GetObject(db.DimStyleTableId, Teigha.DatabaseServices.OpenMode.ForRead) as Teigha.DatabaseServices.DimStyleTable; // Находим в таблице размерных стилей ObjectId нужного стиля по его названию Teigha.DatabaseServices.ObjectId styleId = dimStyleTable["Standard"]; // Открываем таблицу размерного стиля для изменений Teigha.DatabaseServices.DimStyleTableRecord obj = trans.GetObject(styleId, Teigha.DatabaseServices.OpenMode.ForWrite) as Teigha.DatabaseServices.DimStyleTableRecord; ed.WriteMessage($"nDimclrt before: {obj.Dimclrt}"); // Изменяем цвет размерного текста на красный obj.Dimclrt = Teigha.Colors.Color.FromColorIndex(Teigha.Colors.ColorMethod.ByColor, 1); ed.WriteMessage($"nDimclrt after: {obj.Dimclrt}"); trans.Commit(); } } }
